#7 Write a Python function to get the Largest number from a list. 


def max_num_in_list(list):
    max=list[0]
    for a in list:
        if a>max:
            max=a
    return max
print(max_num_in_list([1,2,-8,0]))

#8 Write a Python function to sum all the item in a list.


def sum_list(items):
    sum_numbers=0
    for x in items:
        sum_numbers+=x
    return sum_numbers
print(sum_list([1,2,-8]))
